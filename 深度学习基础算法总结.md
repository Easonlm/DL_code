## <a name="cnn"></a>1. CNN - 卷积神经网络（Convolutional Neural Network）

### 原理
CNN 通过卷积操作提取图像的局部特征，构成多层特征层次。主要成分包括：
- **卷积层**：通过卷积核进行特征提取
- **池化层**：降维，保留重要特征
- **全连接层**：进行分类或回归

### 基础卷积操作代码

```python
import numpy as np

def conv2d(input, kernel, stride=1, padding=0):
    """
    二维卷积操作
    input: (H, W) - 输入矩阵
    kernel: (kH, kW) - 卷积核
    stride: 步幅
    padding: 填充
    """
    H, W = input.shape
    kH, kW = kernel.shape
    
    # 边缘填充
    padded = np.pad(input, ((padding, padding), (padding, padding)))
    
    # 计算输出尺寸
    out_H = (H + 2*padding - kH)//stride + 1
    out_W = (W + 2*padding - kW)//stride + 1
    output = np.zeros((out_H, out_W))
    
    # 卷积计算
    for i in range(out_H):
        for j in range(out_W):
            region = padded[i*stride:i*stride+kH, j*stride:j*stride+kW]
            output[i, j] = np.sum(region * kernel)
    
    return output
```

### 多通道卷积代码

```python
def conv2d_multi_channel(input, kernels, stride=1, padding=0):
    """
    多通道卷积
    input: (in_channels, H, W) - 输入
    kernels: (out_channels, in_channels, kH, kW) - 卷积核组
    """
    in_channels, H, W = input.shape
    out_channels, _, kH, kW = kernels.shape
    
    # 填充每个通道
    padded = np.pad(input, ((0, 0), (padding, padding), (padding, padding)))
    
    out_H = (H + 2*padding - kH)//stride + 1
    out_W = (W + 2*padding - kW)//stride + 1
    output = np.zeros((out_channels, out_H, out_W))
    
    for oc in range(out_channels):
        for i in range(out_H):
            for j in range(out_W):
                val = 0
                for ic in range(in_channels):
                    region = padded[ic, i*stride:i*stride+kH, j*stride:j*stride+kW]
                    val += np.sum(region * kernels[oc, ic])
                output[oc, i, j] = val
    
    return output
```

### 池化操作代码

```python
def max_pool2d(input, pool_size=2, stride=2):
    """
    最大池化
    input: (channels, H, W)
    pool_size: 池化窗口大小
    stride: 池化步幅
    """
    channels, H, W = input.shape
    out_H = (H - pool_size)//stride + 1
    out_W = (W - pool_size)//stride + 1
    output = np.zeros((channels, out_H, out_W))
    
    for c in range(channels):
        for i in range(out_H):
            for j in range(out_W):
                region = input[c, i*stride:i*stride+pool_size, j*stride:j*stride+pool_size]
                output[c, i, j] = np.max(region)
    
    return output

def avg_pool2d(input, pool_size=2, stride=2):
    """平均池化"""
    channels, H, W = input.shape
    out_H = (H - pool_size)//stride + 1
    out_W = (W - pool_size)//stride + 1
    output = np.zeros((channels, out_H, out_W))
    
    for c in range(channels):
        for i in range(out_H):
            for j in range(out_W):
                region = input[c, i*stride:i*stride+pool_size, j*stride:j*stride+pool_size]
                output[c, i, j] = np.mean(region)
    
    return output
```

```python
import torch
import torch.nn as nn

class SimpleCNN(nn.Module):
    def __init__(self, num_classes=10):
        super(SimpleCNN, self).__init__()
        self.conv1 = nn.Conv2d(3, 32, kernel_size=3, padding=1)
        self.pool = nn.MaxPool2d(2, 2)
        self.conv2 = nn.Conv2d(32, 64, kernel_size=3, padding=1)
        self.fc1 = nn.Linear(64 * 8 * 8, 128)
        self.fc2 = nn.Linear(128, num_classes)
        
    def forward(self, x):
        x = self.pool(torch.relu(self.conv1(x)))  # (N, 32, 16, 16)
        x = self.pool(torch.relu(self.conv2(x)))  # (N, 64, 8, 8)
        x = x.view(x.size(0), -1)
        x = torch.relu(self.fc1(x))
        x = self.fc2(x)
        return x
```

## <a name="rnn"></a>2. RNN - 循环神经网络（Recurrent Neural Network）

### 原理
RNN 通过循环连接处理序列数据。隐态 h_t 保存历史信息，与当前输入一起决定下一个隐态。

### 基础 RNN 单元代码

```python
import numpy as np

class RNNCell:
    """基础RNN单元"""
    def __init__(self, input_size, hidden_size):
        self.Wx = np.random.randn(hidden_size, input_size) * 0.01
        self.Wh = np.random.randn(hidden_size, hidden_size) * 0.01
        self.b = np.zeros((hidden_size, 1))
        self.hidden_size = hidden_size
    
    def forward(self, x, h_prev):
        """
        x: (input_size, batch_size) - 当前时刻输入
        h_prev: (hidden_size, batch_size) - 上一时刻隐态
        """
        h = np.tanh(np.dot(self.Wx, x) + np.dot(self.Wh, h_prev) + self.b)
        return h

class RNN:
    """RNN序列处理"""
    def __init__(self, input_size, hidden_size):
        self.rnn_cell = RNNCell(input_size, hidden_size)
        self.hidden_size = hidden_size
    
    def forward(self, X):
        """
        X: (seq_len, batch_size, input_size) - 序列输入
        """
        seq_len, batch_size, _ = X.shape
        h = np.zeros((self.hidden_size, batch_size))
        outputs = []
        
        for t in range(seq_len):
            h = self.rnn_cell.forward(X[t], h)
            outputs.append(h)
        
        return np.array(outputs), h
```

### PyTorch 实现示例

```python
import torch
import torch.nn as nn

class SimpleRNN(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers=1, output_size=1):
        super(SimpleRNN, self).__init__()
        self.rnn = nn.RNN(input_size, hidden_size, num_layers, batch_first=True)
        self.fc = nn.Linear(hidden_size, output_size)
    
    def forward(self, x):
        """
        x: (batch_size, seq_len, input_size)
        """
        out, h_n = self.rnn(x)  # out: (batch_size, seq_len, hidden_size)
        out = self.fc(out[:, -1, :])  # 取最后一个时刻的输出
        return out
```

## <a name="lstm"></a>3. LSTM - 长短期记忆网络（Long Short-Term Memory）

### 原理
LSTM 通过"门"机制解决 RNN 梯度消失问题：
- **遗忘门（Forget Gate）**：决定丢弃多少过去信息
- **输入门（Input Gate）**：决定输入多少新信息
- **输出门（Output Gate）**：决定输出多少内部状态
- **细胞状态（Cell State）**：长期记忆

### LSTM 单元数学公式

```
f_t = σ(W_f · [h_{t-1}, x_t] + b_f)           # 遗忘门
i_t = σ(W_i · [h_{t-1}, x_t] + b_i)           # 输入门
C̃_t = tanh(W_c · [h_{t-1}, x_t] + b_c)       # 候选记忆
C_t = f_t ⊙ C_{t-1} + i_t ⊙ C̃_t             # 细胞状态
o_t = σ(W_o · [h_{t-1}, x_t] + b_o)           # 输出门
h_t = o_t ⊙ tanh(C_t)                         # 隐态输出
```

### 基础 LSTM 实现

```python
import numpy as np

class LSTMCell:
    """LSTM单元"""
    def __init__(self, input_size, hidden_size):
        # 遗忘门、输入门、候选、输出门的权重
        self.W = np.random.randn(4*hidden_size, input_size + hidden_size) * 0.01
        self.b = np.zeros((4*hidden_size, 1))
        self.hidden_size = hidden_size
    
    def forward(self, x, h_prev, c_prev):
        """
        x: (input_size, batch_size)
        h_prev: (hidden_size, batch_size)
        c_prev: (hidden_size, batch_size)
        """
        # 连接输入和隐态
        z = np.vstack([h_prev, x])
        
        # 计算四个门
        gates = np.dot(self.W, z) + self.b
        
        # 分割门输出
        f = 1 / (1 + np.exp(-gates[:self.hidden_size]))        # sigmoid, 遗忘门
        i = 1 / (1 + np.exp(-gates[self.hidden_size:2*self.hidden_size]))  # sigmoid, 输入门
        c_tilde = np.tanh(gates[2*self.hidden_size:3*self.hidden_size])    # tanh, 候选
        o = 1 / (1 + np.exp(-gates[3*self.hidden_size:]))      # sigmoid, 输出门
        
        # 更新细胞状态和隐态
        c = f * c_prev + i * c_tilde
        h = o * np.tanh(c)
        
        return h, c

class LSTM:
    """LSTM序列处理"""
    def __init__(self, input_size, hidden_size):
        self.lstm_cell = LSTMCell(input_size, hidden_size)
        self.hidden_size = hidden_size
    
    def forward(self, X):
        """X: (seq_len, batch_size, input_size)"""
        seq_len, batch_size, _ = X.shape
        h = np.zeros((self.hidden_size, batch_size))
        c = np.zeros((self.hidden_size, batch_size))
        outputs = []
        
        for t in range(seq_len):
            h, c = self.lstm_cell.forward(X[t], h, c)
            outputs.append(h)
        
        return np.array(outputs), (h, c)
```

```python
    def __init__(self, input_size, hidden_size, num_layers=1, output_size=1):
```

## <a name="gru"></a>4. GRU - 门控循环单元（Gated Recurrent Unit）

### 原理
GRU 是 LSTM 的简化版，只有两个门：
- **重置门（Reset Gate）**：控制过去信息流
- **更新门（Update Gate）**：控制新旧信息的混合比例

### GRU 数学公式

```
r_t = σ(W_r · [h_{t-1}, x_t] + b_r)           # 重置门
u_t = σ(W_u · [h_{t-1}, x_t] + b_u)           # 更新门
h̃_t = tanh(W_h · [r_t ⊙ h_{t-1}, x_t] + b_h) # 候选隐态
h_t = (1 - u_t) ⊙ h_{t-1} + u_t ⊙ h̃_t        # 隐态更新
```

### 基础 GRU 实现

```python
import numpy as np

class GRUCell:
    """GRU单元"""
    def __init__(self, input_size, hidden_size):
        self.W = np.random.randn(3*hidden_size, input_size + hidden_size) * 0.01
        self.b = np.zeros((3*hidden_size, 1))
        self.hidden_size = hidden_size
    
    def forward(self, x, h_prev):
        """
        x: (input_size, batch_size)
        h_prev: (hidden_size, batch_size)
        """
        z = np.vstack([h_prev, x])
        gates = np.dot(self.W, z) + self.b
        
        # 重置门和更新门
        r = 1 / (1 + np.exp(-gates[:self.hidden_size]))
        u = 1 / (1 + np.exp(-gates[self.hidden_size:2*self.hidden_size]))
        
        # 候选隐态
        h_tilde = np.tanh(gates[2*self.hidden_size:] + 
                         np.dot(r, h_prev))
        
        # 隐态更新
        h = (1 - u) * h_prev + u * h_tilde
        
        return h

class GRU:
    """GRU序列处理"""
    def __init__(self, input_size, hidden_size):
        self.gru_cell = GRUCell(input_size, hidden_size)
        self.hidden_size = hidden_size
    
    def forward(self, X):
        """X: (seq_len, batch_size, input_size)"""
        seq_len, batch_size, _ = X.shape
        h = np.zeros((self.hidden_size, batch_size))
        outputs = []
        
        for t in range(seq_len):
            h = self.gru_cell.forward(X[t], h)
            outputs.append(h)
        
        return np.array(outputs), h
```

## <a name="transformer"></a>5. Transformer - 转换器

### 原理
Transformer 使用"自注意力机制"替代循环，具备并行计算能力：
- **多头自注意力（Multi-Head Self-Attention）**：捕捉不同方面的依赖关系
- **前馈网络（Feed-Forward Network）**：增加非线性
- **位置编码（Positional Encoding）**：注入序列位置信息
- **分层堆叠**：多层 Encoder 和 Decoder

### 注意力机制公式

```
Attention(Q, K, V) = softmax(Q·K^T / √d_k)·V

其中：
Q = 查询向量 (Query)
K = 键向量 (Key)
V = 值向量 (Value)
d_k = 键的维度
```

### 基础注意力函数

```python
import numpy as np

def softmax(x):
    """softmax激活函数"""
    e_x = np.exp(x - np.max(x, axis=-1, keepdims=True))
    return e_x / np.sum(e_x, axis=-1, keepdims=True)

def scaled_dot_product_attention(Q, K, V, mask=None):
    """
    缩放点积注意力
    Q: (batch_size, num_heads, seq_len, d_k)
    K: (batch_size, num_heads, seq_len, d_k)
    V: (batch_size, num_heads, seq_len, d_v)
    """
    d_k = K.shape[-1]
    
    # 计算注意力分数
    scores = np.matmul(Q, K.transpose(0, 1, 3, 2)) / np.sqrt(d_k)
    
    if mask is not None:
        scores = scores + (mask * -1e9)
    
    # softmax 得到注意力权重
    attention_weights = softmax(scores)
    
    # 加权得到输出
    output = np.matmul(attention_weights, V)
    
    return output, attention_weights
```

### 位置编码

```python
def positional_encoding(seq_len, d_model):
    """
    生成位置编码
    seq_len: 序列长度
    d_model: 模型维度
    """
    pe = np.zeros((seq_len, d_model))
    position = np.arange(0, seq_len, dtype=np.float32).reshape(-1, 1)
    div_term = np.exp(np.arange(0, d_model, 2) * -(np.log(10000.0) / d_model))
    
    pe[:, 0::2] = np.sin(position * div_term)
    pe[:, 1::2] = np.cos(position * div_term)
    
    return pe
```

### PyTorch 实现示例

```python
import torch
import torch.nn as nn
import math

class MultiHeadAttention(nn.Module):
    def __init__(self, d_model, num_heads):
        super(MultiHeadAttention, self).__init__()
        assert d_model % num_heads == 0
        
        self.d_model = d_model
        self.num_heads = num_heads
        self.d_k = d_model // num_heads
        
        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)
        self.W_o = nn.Linear(d_model, d_model)
    
    def forward(self, Q, K, V, mask=None):
        batch_size = Q.shape[0]
        
        # 线性变换并分头
        Q = self.W_q(Q).view(batch_size, -1, self.num_heads, self.d_k).transpose(1, 2)
        K = self.W_k(K).view(batch_size, -1, self.num_heads, self.d_k).transpose(1, 2)
        V = self.W_v(V).view(batch_size, -1, self.num_heads, self.d_k).transpose(1, 2)
        
        # 缩放点积注意力
        scores = torch.matmul(Q, K.transpose(-2, -1)) / math.sqrt(self.d_k)
        if mask is not None:
            scores = scores.masked_fill(mask == 0, -1e9)
        
        attention_weights = torch.softmax(scores, dim=-1)
        context = torch.matmul(attention_weights, V)
        
        # 合并多头
        context = context.transpose(1, 2).contiguous().view(batch_size, -1, self.d_model)
        
        # 最后的线性变换
        output = self.W_o(context)
        
        return output, attention_weights

class TransformerEncoderLayer(nn.Module):
    def __init__(self, d_model, num_heads, d_ff, dropout=0.1):
        super(TransformerEncoderLayer, self).__init__()
        self.attention = MultiHeadAttention(d_model, num_heads)
        self.feed_forward = nn.Sequential(
            nn.Linear(d_model, d_ff),
            nn.ReLU(),
            nn.Linear(d_ff, d_model)
        )
        self.norm1 = nn.LayerNorm(d_model)
        self.norm2 = nn.LayerNorm(d_model)
        self.dropout = nn.Dropout(dropout)
    
    def forward(self, x, mask=None):
        # 自注意力
        attn_output, _ = self.attention(x, x, x, mask)
        x = self.norm1(x + self.dropout(attn_output))
        
        # 前馈网络
        ff_output = self.feed_forward(x)
        x = self.norm2(x + self.dropout(ff_output))
        
        return x

class SimpleTransformer(nn.Module):
    def __init__(self, vocab_size, d_model, num_heads, num_layers, d_ff, max_seq_len):
        super(SimpleTransformer, self).__init__()
        self.d_model = d_model
        
        self.embedding = nn.Embedding(vocab_size, d_model)
        self.positional_encoding = self._positional_encoding(max_seq_len, d_model)
        
        self.encoder_layers = nn.ModuleList([
            TransformerEncoderLayer(d_model, num_heads, d_ff)
            for _ in range(num_layers)
        ])
        
        self.fc = nn.Linear(d_model, vocab_size)
    
    def _positional_encoding(self, max_seq_len, d_model):
        pe = torch.zeros(max_seq_len, d_model)
        position = torch.arange(0, max_seq_len, dtype=torch.float).unsqueeze(1)
        div_term = torch.exp(torch.arange(0, d_model, 2).float() * -(math.log(10000.0) / d_model))
        
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        
        return pe.unsqueeze(0)
    
    def forward(self, x):
        # 嵌入和位置编码
        x = self.embedding(x) * math.sqrt(self.d_model)
        x = x + self.positional_encoding[:, :x.size(1), :].to(x.device)
        
        # 通过编码层
        for encoder in self.encoder_layers:
            x = encoder(x)
        
        # 输出层
        x = self.fc(x)
        
        return x
```

## <a name="other"></a>6. 其他重要算法

### 6.1 批量归一化（Batch Normalization）

```python
class BatchNorm1d:
    """一维批量归一化"""
    def __init__(self, num_features, eps=1e-5, momentum=0.1):
        self.eps = eps
        self.momentum = momentum
        self.gamma = np.ones(num_features)
        self.beta = np.zeros(num_features)
        self.running_mean = np.zeros(num_features)
        self.running_var = np.ones(num_features)
    
    def forward(self, x, training=True):
        if training:
            batch_mean = np.mean(x, axis=0)
            batch_var = np.var(x, axis=0)
            
            x_norm = (x - batch_mean) / np.sqrt(batch_var + self.eps)
            
            self.running_mean = self.momentum * batch_mean + (1 - self.momentum) * self.running_mean
            self.running_var = self.momentum * batch_var + (1 - self.momentum) * self.running_var
        else:
            x_norm = (x - self.running_mean) / np.sqrt(self.running_var + self.eps)
        
        return self.gamma * x_norm + self.beta
```

### 6.2 Dropout 正则化

```python
class Dropout:
    """Dropout层"""
    def __init__(self, dropout_rate=0.5):
        self.dropout_rate = dropout_rate
    
    def forward(self, x, training=True):
        if training:
            mask = np.random.binomial(1, 1 - self.dropout_rate, x.shape)
            return x * mask / (1 - self.dropout_rate)
        else:
            return x
```

### 6.3 激活函数

```python
def relu(x):
    return np.maximum(0, x)

def sigmoid(x):
    return 1 / (1 + np.exp(-x))

def tanh(x):
    return np.tanh(x)

def softmax(x):
    e_x = np.exp(x - np.max(x, axis=-1, keepdims=True))
    return e_x / np.sum(e_x, axis=-1, keepdims=True)

def gelu(x):
    """GELU激活函数 - Transformer常用"""
    cdf = 0.5 * (1.0 + np.tanh(np.sqrt(2 / np.pi) * (x + 0.044715 * x**3)))
    return x * cdf
```

```python
def cross_entropy_loss(y_true, y_pred):
    """交叉熵损失"""
    batch_size = y_true.shape[0]
    y_pred_clipped = np.clip(y_pred, 1e-7, 1 - 1e-7)
    loss = -np.sum(y_true * np.log(y_pred_clipped)) / batch_size
    return loss

def mse_loss(y_true, y_pred):
    """均方误差损失"""
    return np.mean((y_true - y_pred) ** 2)

def l1_loss(y_true, y_pred):
    """L1损失"""
    return np.mean(np.abs(y_true - y_pred))
```

### 6.5 优化器

```python
class SGD:
    """随机梯度下降"""
    def __init__(self, params, lr=0.01):
        self.params = params
        self.lr = lr
    
    def step(self, grads):
        for param, grad in zip(self.params, grads):
            param -= self.lr * grad

class Adam:
    """Adam优化器"""
    def __init__(self, params, lr=0.001, beta1=0.9, beta2=0.999, eps=1e-8):
        self.params = params
        self.lr = lr
        self.beta1 = beta1
        self.beta2 = beta2
        self.eps = eps
        self.m = [np.zeros_like(p) for p in params]  # 一阶矩
        self.v = [np.zeros_like(p) for p in params]  # 二阶矩
        self.t = 0
    
    def step(self, grads):
        self.t += 1
        for i, (param, grad) in enumerate(zip(self.params, grads)):
            self.m[i] = self.beta1 * self.m[i] + (1 - self.beta1) * grad
            self.v[i] = self.beta2 * self.v[i] + (1 - self.beta2) * grad**2
            
            m_hat = self.m[i] / (1 - self.beta1**self.t)
            v_hat = self.v[i] / (1 - self.beta2**self.t)
            
            param -= self.lr * m_hat / (np.sqrt(v_hat) + self.eps)
```

## 总结对比表

| 算法 | 输入类型 | 优点 | 缺点 | 应用 |
|------|--------|------|------|------|
| CNN | 图像 | 局部特征提取，参数少 | 难以处理序列 | 图像分类、目标检测 |
| RNN | 序列 | 处理序列，保存历史 | 梯度消失/爆炸 | 文本、时间序列 |
| LSTM | 序列 | 解决梯度问题，长期依赖 | 计算复杂 | NLP、时间序列预测 |
| GRU | 序列 | 参数少于LSTM，计算快 | 表达力略弱 | NLP、序列分析 |
| Transformer | 序列 | 并行计算，捕捉全局依赖 | 内存消耗大 | NLP、翻译、预训练 |

```python
import torch
import torch.nn as nn
from torch.optim import Adam
from torch.utils.data import DataLoader

def train_epoch(model, train_loader, criterion, optimizer, device):
    model.train()
    total_loss = 0
    
    for batch_idx, (data, target) in enumerate(train_loader):
        data, target = data.to(device), target.to(device)
        
        # 前向传播
        output = model(data)
        loss = criterion(output, target)
        
        # 反向传播
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        
        total_loss += loss.item()
    
    return total_loss / len(train_loader)

def evaluate(model, test_loader, criterion, device):
    model.eval()
    total_loss = 0
    correct = 0
    
    with torch.no_grad():
        for data, target in test_loader:
            data, target = data.to(device), target.to(device)
            output = model(data)
            loss = criterion(output, target)
            total_loss += loss.item()
            
            # 计算准确率
            pred = output.argmax(dim=1)
            correct += pred.eq(target).sum().item()
    
    accuracy = correct / len(test_loader.dataset)
    avg_loss = total_loss / len(test_loader)
    
    return avg_loss, accuracy
```
